def self.new(id, level, code, reason, classid = 0, methodid = 0)
  case level
  when :connection
    ConnectionClosed.new(code, reason, classid, methodid)
  when :channel
    ChannelClosed.new(id, code, reason, classid, methodid)
  else raise ArgumentError, "invalid level '#{level}'"
  end
end
